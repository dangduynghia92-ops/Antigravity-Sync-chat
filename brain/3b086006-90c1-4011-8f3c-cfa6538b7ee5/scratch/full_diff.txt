+def apply_chapter_splits(phase_plan: dict, log_callback=None) -> dict:
+    """Deterministic chapter splitting based on final node_count per phase.
+
+    Rules:
+      - Hook phase (first) О“ГҐГ† always 1 chapter
+      - End phase (last)   О“ГҐГ† always 1 chapter
+      - Body phases with 0 main_key_data О“ГҐГ† MERGE into nearest adjacent phase
+      - Body phases:
+          1-3 main_key_data О“ГҐГ† 1 chapter
+          4-6 main_key_data О“ГҐГ† 2 chapters
+          7+  main_key_data О“ГҐГ† 3 chapters
+
+    Assigns sequential chapter numbers and updates total_chapters.
+    """
+    phases = phase_plan.get("phase_chapter_plan", [])
+    if not phases:
+        return phase_plan
+
+    # О“Г¶Г‡О“Г¶Г‡ Step 1: Merge empty body phases into neighbors О“Г¶Г‡О“Г¶Г‡
+    merged_indices = set()
+    for idx in range(1, len(phases) - 1):  # skip hook (0) and end (last)
+        phase = phases[idx]
+        node_count = len(phase.get("main_key_data", []))
+        if node_count == 0:
+            # Find best neighbor to merge into (prefer next, fallback prev)
+            target_idx = idx + 1 if idx + 1 < len(phases) - 1 else idx - 1
+            if target_idx < 1:
+                target_idx = 1  # never merge into hook
+            target = phases[target_idx]
+
+            # Move sub_key_data to target
+            target.setdefault("sub_key_data", []).extend(
+                phase.get("sub_key_data", [])
+            )
+            merged_indices.add(idx)
+            if log_callback:
+                log_callback(
+                    f"  [MERGE] {phase.get('phase', '?')} (0 main) О“ГҐГ† merged into {target.get('phase', '?')}"
+                )
+
+    if merged_indices:
+        phases = [p for i, p in enumerate(phases) if i not in merged_indices]
+        phase_plan["phase_chapter_plan"] = phases
+
+    # О“Г¶Г‡О“Г¶Г‡ Step 2: Assign chapters О“Г¶Г‡О“Г¶Г‡
+    chapter_num = 1
+    total_chapters = 0
+
+    for idx, phase in enumerate(phases):
+        is_hook = (idx == 0)
+        is_end = (idx == len(phases) - 1)
+        node_count = len(phase.get("main_key_data", []))
+        phase["node_count"] = node_count
+
+        if is_hook or is_end:
+            num_chapters = 1
+        elif node_count <= 3:
+            num_chapters = 1
+        elif node_count <= 6:
+            num_chapters = 2
+        else:
+            num_chapters = 3
+
+        chapters = list(range(chapter_num, chapter_num + num_chapters))
+        phase["chapters"] = chapters
+
+        if num_chapters > 1:
+            phase["capacity_check"] = f"{node_count} main > {3 if num_chapters == 2 else 6} О“ГҐГ† SPLIT into {num_chapters} chapters"
+            phase["reason_for_split"] = f"{node_count} main events across {num_chapters} chapters"
+        else:
+            phase["capacity_check"] = f"{node_count} main О“Г«Г± 3 О“ГҐГ† KEEP"
+            phase["reason_for_split"] = None
+
+        action = "SPLIT" if num_chapters > 1 else "KEEP"
+        if log_callback:
+            log_callback(
+                f"  [{action}] {phase.get('phase', '?')}: ch{chapters} ({node_count} main)"
+            )
+
+        chapter_num += num_chapters
+        total_chapters += num_chapters
+
+    phase_plan["total_chapters"] = total_chapters
+
+    if log_callback:
+        log_callback(f"  Total: {total_chapters} chapters across {len(phases)} phases")
+
+    return phase_plan

+def _resolve_tier(raw_value: str, tier_scoring: dict) -> tuple:
+    """Match a raw blueprint tier value to its score.
+
+    Returns: (matched_tier_name, score, match_type)
+    match_type: "exact" | "fuzzy" | "none"
+    """
+    if not raw_value or not isinstance(raw_value, str):
+        return None, 0, "none"
+
+    # 1. Exact match
+    if raw_value in tier_scoring:
+        return raw_value, tier_scoring[raw_value], "exact"
+
+    # 2. Case-insensitive exact match
+    for tier_name, score in tier_scoring.items():
+        if raw_value.strip().lower() == tier_name.lower():
+            return tier_name, score, "exact"

+    # 3. Fuzzy keyword match
+    raw_lower = raw_value.lower()

+    best_score = 0
+    for tier_name, keywords in _TIER_FUZZY_MAP.items():
+        if tier_name not in tier_scoring:
